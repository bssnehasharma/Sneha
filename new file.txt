import streamlit as st
import pandas as pd 
import plotly.express as px 

# Connect to MySQL 
@st.cache_data
def get_data():
    df = pd.read_csv("Ola_Dataset.csv")

    numeric_cols = ['Booking_Value', 'Ride_Distance', 'Driver_Ratings', 'Customer_Rating', 'V_TAT', 'C_TAT']
    for col in numeric_cols:
        if col in df.columns:
            df[col] = pd.to_numeric(df[col], errors='coerce')

    if 'Date' in df.columns:
        df['Date'] = pd.to_datetime(df['Date'], errors='coerce')

    return df

st.title("Ola Rides Analysis 🚖")

# Load data
df = get_data()
numeric_cols = ['Booking_Value', 'Ride_Distance', 'Driver_Ratings', 'Customer_Rating', 'V_TAT', 'C_TAT']
for col in numeric_cols:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col], errors='coerce')

# Sidebar filters
st.sidebar.header("Filters")
status_list = df['Booking_Status'].unique()
status = st.sidebar.multiselect("Booking Status", status_list, default=status_list)

vehicle_list = df['Vehicle_Type'].unique()
vehicle = st.sidebar.multiselect("Vehicle Type", vehicle_list, default=vehicle_list)

df = df[df['Booking_Status'].isin(status) & df['Vehicle_Type'].isin(vehicle)]

# Show metrics
st.subheader("Key Metrics")
st.write(f"Total Rides: {len(df):,}")
st.write(f"Success Rides: {len(df[df['Booking_Status'] == 'Success']):,}")
st.write(f"Avg Booking Value: ₹{df['Booking_Value'].mean():.0f}")

# Show charts - using only built-in Streamlit charts, no Plotly needed
st.subheader("Rides by Vehicle Type")
st.bar_chart(df['Vehicle_Type'].value_counts())

st.subheader("Payment Method Count")
st.bar_chart(df['Payment_Method'].value_counts())

# Show data
st.subheader("Raw Data")
st.dataframe(df)
